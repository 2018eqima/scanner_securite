# api-scanner.eqima.org — Backend Spring Boot EQIMA Scanner

server {
    listen 80;
    server_name api-scanner.eqima.org;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name api-scanner.eqima.org;

    ssl_certificate     /etc/letsencrypt/live/api-scanner.eqima.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api-scanner.eqima.org/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    client_max_body_size 10M;

    location / {
        proxy_pass            http://127.0.0.1:8088;
        proxy_set_header      Host $host;
        proxy_set_header      X-Real-IP $remote_addr;
        proxy_set_header      X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header      X-Forwarded-Proto $scheme;
        proxy_read_timeout    300s;
        proxy_send_timeout    300s;

        # SSE — désactiver le buffering pour les événements live
        proxy_buffering       off;
        proxy_cache           off;
        proxy_set_header      Connection '';
        proxy_http_version    1.1;
    }
}
